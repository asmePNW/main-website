//every week
SELECT cron.schedule(
  'archive-past-presidents',
  '0 0 * * 1',
  $$
  SELECT net.http_get(
    url := 'example.com/api/[slug]/notify?secret=insert_secret',
    timeout_milliseconds := 30000
  );
  $$
);


//Every 6 hours (ie: 00:00, 06:00, 12:00, 18:00 UTC)
SELECT cron.schedule(
  'notify-new-events',
  '0 */6 * * *',
  $$
  SELECT net.http_get(
    url := 'example.com/api/[slug]/notify?secret=insert_secret',
    timeout_milliseconds := 30000
  );
  $$
);
